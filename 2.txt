```python
from typing import List, Tuple, Optional, Dict
import numpy as np
from sklearn.feature_extraction.text import TfidfVectorizer

def extract_text_features(
    variations: List[Dict],
    vectorizer: Optional[TfidfVectorizer] = None,
    max_features: int = 5000
) -> Tuple[np.ndarray, TfidfVectorizer]:
    """
    Convert a list of variations into TF-IDF feature vectors.
    Returns (feature_matrix, vectorizer).
    """
    contents = [v.get("content", "") for v in variations]
    if vectorizer is None:
        vectorizer = TfidfVectorizer(max_features=max_features, stop_words="english")
        X = vectorizer.fit_transform(contents)
    else:
        X = vectorizer.transform(contents)
    return X.toarray(), vectorizer
```

#### `backend/entropy_analysis/entropy.py`
```python
from typing import List
import numpy as np
from scipy.stats import entropy as scipy_entropy

EPS = 1e-12

def _to_probability(vecs: np.ndarray, eps: float = EPS) -> np.ndarray:
    """
    Aggregate per-item mass and normalize to a probability distribution.
    vecs: shape (n_items, n_features)
    """
    if vecs.size == 0:
        return np.array([])
    masses = np.maximum(vecs.sum(axis=1), eps)
    total = masses.sum()
    if total <= 0:
        return np.ones_like(masses) / len(masses)
    return masses / total

def system_entropy(feature_vectors: np.ndarray) -> float:
    """
    Compute Shannon entropy of aggregated feature masses.
    """
    probs = _to_probability(feature_vectors)
    if probs.size == 0:
        return 0.0
    return float(scipy_entropy(probs, base=np.e))

def leave_one_out_deltas(feature_vectors: np.ndarray) -> List[float]:
    """
    Compute H - H_{-i} for each item i.
    Returns list of delta entropies in same order as rows.
    """
    if feature_vectors.shape[0] == 0:
        return []
    H = system_entropy(feature_vectors)
    deltas = []
    n = feature_vectors.shape[0]
    for i in range(n):
        reduced = np.delete(feature_vectors, i, axis=0)
        H_minus = system_entropy(reduced)
        deltas.append(float(H - H_minus))
    return deltas
```

#### `backend/entropy_analysis/neo4j_writer.py`
```python
from neo4j import GraphDatabase
from typing import List, Dict

class Neo4jWriter:
    def __init__(self, uri: str, user: str, password: str):
        self.driver = GraphDatabase.driver(uri, auth=(user, password))

    def close(self):
        self.driver.close()

    def write_entropy_scores(self, records: List[Dict]) -> int:
        """
        records: list of {id, entropy_score, anomaly_flag}
        Returns number of updated nodes.
        """
        query = """
        UNWIND $rows AS r
        MATCH (v:Variation {id: r.id})
        SET v.entropy_score = r.entropy_score,
            v.anomaly_flag = r.anomaly_flag
        RETURN count(v) AS updated
        """
        with self.driver.session() as session:
            result = session.run(query, rows=records)
            row = result.single()
            return int(row["updated"]) if row else 0
```

#### `backend/entropy_analysis/bayesian.py`
```python
from typing import Dict, List
import math

def normalize_scores(scores: List[float]) -> List[float]:
    total = sum(scores)
    if total == 0:
        n = len(scores)
        return [1.0 / n] * n if n > 0 else []
    return [s / total for s in scores]

def originality_score(consistency: float, resolution: float, reputation: float, entropy_score: float, weights: Dict = None) -> float:
    """
    Compose OriginalityScore using weights:
    default weights: consistency 0.35, resolution 0.35, reputation 0.20, entropy 0.10 (inverted)
    entropy_score should be normalized to [0,1] where lower is better (original).
    """
    if weights is None:
        weights = {"consistency":0.35, "resolution":0.35, "reputation":0.20, "entropy":0.10}
    inv_entropy = 1.0 - entropy_score
    score = (weights["consistency"] * consistency +
             weights["resolution"] * resolution +
             weights["reputation"] * reputation +
             weights["entropy"] * inv_entropy)
    return score
```

#### `backend/entropy_analysis/api.py`
```python
from typing import List, Dict
from .features import extract_text_features
from .entropy import leave_one_out_deltas
from .neo4j_writer import Neo4jWriter

def classify_anomaly(norm_delta: float, threshold: float = 0.05) -> str:
    return "ANOMALY" if norm_delta >= threshold else "STABLE"

def run_entropy_analysis(variations: List[Dict], neo4j_config: Dict) -> Dict:
    """
    Orchestrate feature extraction, entropy scoring, and Neo4j writes.
    """
    features, vectorizer = extract_text_features(variations)
    deltas = leave_one_out_deltas(features)
    max_delta = max(deltas) if deltas else 1.0
    records = []
    for v, d in zip(variations, deltas):
        norm = d / max_delta if max_delta > 0 else 0.0
        flag = classify_anomaly(norm)
        records.append({
            "id": v["id"],
            "entropy_score": float(d),
            "anomaly_flag": flag
        })
    writer = Neo4jWriter(neo4j_config["uri"], neo4j_config["user"], neo4j_config["password"])
    updated = writer.write_entropy_scores(records)
    writer.close()
    return {"updated": updated, "count": len(records)}
```

```python
from fastapi import FastAPI, HTTPException
from pydantic import BaseModel
from typing import List
from entropy_analysis.api import run_entropy_analysis

app = FastAPI(title="Entropy Analysis API")

class VariationIn(BaseModel):
    id: str
    content: str

class Neo4jConfig(BaseModel):
    uri: str
    user: str
    password: str

class AnalysisRequest(BaseModel):
    variations: List[VariationIn]
    neo4j: Neo4jConfig

@app.post("/entropy-analysis")
async def entropy_analysis(req: AnalysisRequest):
    try:
        variations = [v.dict() for v in req.variations]
        neo4j_cfg = req.neo4j.dict()
        result = run_entropy_analysis(variations, neo4j_cfg)
        return {"status": "ok", "result": result}
    except Exception as e:
        raise HTTPException(status_code=500, detail=str(e))

@app.get("/status")
async def status():
    return {"status": "ready"}
```

#### `scripts/networkx_tools.py`
```python
import networkx as nx

def build_graph_from_edges(edges):
    """
    edges: list of (from_id, to_id) representing EVOLVED_FROM or similar
    """
    G = nx.DiGraph()
    G.add_edges_from(edges)
    return G

def topological_roots(G):
    """
    Return nodes with zero in-degree (candidate roots).
    """
    return [n for n, d in G.in_degree() if d == 0]

def reach_scores(G):
    """
    Compute reachability-based score for each node (number of descendants).
    """
    scores = {}
    for n in G.nodes():
        scores[n] = len(nx.descendants(G, n))
    return scores
```

#### `docs/bayesian_scoring.cypher`
```cypher
// Example: compute normalized OriginalityScore and persist
MATCH (v:Variation)
WHERE exists(v.consistency) AND exists(v.resolution_power) AND exists(v.reputation) AND exists(v.entropy_score)
WITH v,
     v.consistency AS consistency,
     v.resolution_power AS resolution,
     v.reputation AS reputation,
     v.entropy_score AS entropy
WITH v, (0.35*consistency + 0.35*resolution + 0.20*reputation + 0.10*(1.0 - entropy)) AS originality
SET v.originality_score = originality
RETURN v.id, v.originality_score
```
